Waveform reale IV.LAV9

Scarica 60 secondi di segnale miniSEED da INGV e lo disegna con ObsPy.

Pronto.
packages = [ "numpy", "matplotlib", "obspy" ] from pyscript import document from pyodide.http import pyfetch from io import BytesIO import numpy as np import matplotlib matplotlib.use("module://matplotlib_pyodide.html5_canvas_backend") import matplotlib.pyplot as plt from obspy import read async def load_lav9(): status = document.getElementById("status") plot_div = document.getElementById("plot") status.innerText = "Scarico il dato reale da INGV..." plot_div.innerHTML = "" # 60 secondi di waveform reale url = ( "https://webservices.ingv.it/fdsnws/dataselect/1/query" "?net=IV" "&sta=LAV9" "&loc=00" "&cha=HHZ" "&starttime=2024-01-01T00:00:00" "&endtime=2024-01-01T00:01:00" ) try: response = await pyfetch(url) if response.status != 200: status.innerText = f"Errore INGV: HTTP {response.status}" return data = await response.bytes() # ObsPy legge il miniSEED da memoria st = read(BytesIO(data), format="MSEED") tr = st[0] tr.detrend("demean") fs = tr.stats.sampling_rate t = np.arange(tr.stats.npts) / fs document.pyodideMplTarget = plot_div plt.figure(figsize=(11, 4)) plt.plot(t, tr.data) plt.title(f"{tr.id} - {tr.stats.starttime}") plt.xlabel("Tempo [s]") plt.ylabel("Ampiezza") plt.grid(True) plt.tight_layout() plt.show() status.innerText = ( "Dato caricato correttamente.\n" f"Trace: {tr.id}\n" f"Start: {tr.stats.starttime}\n" f"End: {tr.stats.endtime}\n" f"Sampling rate: {tr.stats.sampling_rate} Hz\n" f"Campioni: {tr.stats.npts}" ) except Exception as e: status.innerText = ( "Errore durante il caricamento.\n\n" + str(e) + "\n\nPossibile causa: CORS, servizio INGV non disponibile, " "oppure ObsPy non compatibile nel browser." )
